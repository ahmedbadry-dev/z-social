## SYSTEM LOGIC (Instagram-style)

### Public Account (default):
- Any user can follow directly → no approval needed
- Follower can see everything: posts, profile, followers list
- Notification to target: "B started following you"

### Private Account:
- Follow button → sends Follow Request (not immediate follow)
- Target receives notification: "B requested to follow you"
- Target can: Accept → B becomes follower | Reject → nothing happens
- After accept: B can see posts, profile, etc.
- B never gets notified if rejected
- B gets notified if accepted: "A accepted your follow request"
- Profile still shows: name, image, bio, followers count (even if private)
- Profile HIDES: posts list (shows lock message instead)

### Button states:
- Not following, public account → "Follow"
- Not following, private account, no request sent → "Follow" (sends request)
- Request sent, waiting → "Requested" (can cancel)
- Following → "Following" (can unfollow)
- Target follows ME but I don't follow them → "Follow back"

#### followUser mutation (handles both public and private):
```typescript
export const followUser = mutation({
  args: { targetUserId: v.string() },
  handler: async (ctx, args) => {
    const currentUserId = await requireAuthUserId(ctx)
    if (currentUserId === args.targetUserId) {
      throw new ConvexError("You cannot follow yourself")
    }

    // Check if already following
    const existingFollow = await ctx.db
      .query("follows")
      .withIndex("by_pair", (q) =>
        q.eq("followerId", currentUserId).eq("followingId", args.targetUserId)
      )
      .unique()

    if (existingFollow) {
      // Unfollow
      await ctx.db.delete(existingFollow._id)
      return { action: "unfollowed" }
    }

    // Check if target account is private
    const targetProfile = await ctx.db
      .query("userProfiles")
      .withIndex("by_user", (q) => q.eq("userId", args.targetUserId))
      .first()

    const isPrivate = targetProfile?.isPrivate ?? false

    if (isPrivate) {
      // Check if request already exists
      const existingRequest = await ctx.db
        .query("followRequests")
        .withIndex("by_pair", (q) =>
          q.eq("fromUserId", currentUserId).eq("toUserId", args.targetUserId)
        )
        .unique()

      if (existingRequest) {
        if (existingRequest.status === "pending") {
          // Cancel the request
          await ctx.db.delete(existingRequest._id)
          return { action: "request_cancelled" }
        }
      }

      // Send follow request
      await ctx.db.insert("followRequests", {
        fromUserId: currentUserId,
        toUserId: args.targetUserId,
        status: "pending",
        createdAt: Date.now(),
      })

      // Notify target
      await ctx.db.insert("notifications", {
        userId: args.targetUserId,
        actorId: currentUserId,
        type: "follow_request",
        read: false,
        createdAt: Date.now(),
      })

      return { action: "request_sent" }
    } else {
      // Public account — follow directly
      await ctx.db.insert("follows", {
        followerId: currentUserId,
        followingId: args.targetUserId,
        createdAt: Date.now(),
      })

      // Notify target
      await ctx.db.insert("notifications", {
        userId: args.targetUserId,
        actorId: currentUserId,
        type: "follow",
        read: false,
        createdAt: Date.now(),
      })

      return { action: "followed" }
    }
  },
})
```

#### acceptFollowRequest mutation:
```typescript
export const acceptFollowRequest = mutation({
  args: { fromUserId: v.string() },
  handler: async (ctx, args) => {
    const currentUserId = await requireAuthUserId(ctx)

    const request = await ctx.db
      .query("followRequests")
      .withIndex("by_pair", (q) =>
        q.eq("fromUserId", args.fromUserId).eq("toUserId", currentUserId)
      )
      .unique()

    if (!request || request.status !== "pending") {
      throw new ConvexError("Follow request not found")
    }

    // Accept: create follow relationship
    await ctx.db.patch(request._id, { status: "accepted" })

    await ctx.db.insert("follows", {
      followerId: args.fromUserId,
      followingId: currentUserId,
      createdAt: Date.now(),
    })

    // Notify requester that request was accepted
    await ctx.db.insert("notifications", {
      userId: args.fromUserId,
      actorId: currentUserId,
      type: "follow_accept",
      read: false,
      createdAt: Date.now(),
    })

    return { accepted: true }
  },
})
```

```typescript
export const rejectFollowRequest = mutation({
  args: { fromUserId: v.string() },
  handler: async (ctx, args) => {
    const currentUserId = await requireAuthUserId(ctx)

    const request = await ctx.db
      .query("followRequests")
      .withIndex("by_pair", (q) =>
        q.eq("fromUserId", args.fromUserId).eq("toUserId", currentUserId)
      )
      .unique()

    if (!request) throw new ConvexError("Follow request not found")

    await ctx.db.patch(request._id, { status: "rejected" })
    // No notification sent to requester on rejection

    return { rejected: true }
  },
})
```

```typescript
export const getFollowStatus = query({
  args: { targetUserId: v.string() },
  handler: async (ctx, args) => {
    const currentUserId = await getCurrentUserId(ctx)

    const isFollowing = currentUserId
      ? !!(await ctx.db
          .query("follows")
          .withIndex("by_pair", (q) =>
            q.eq("followerId", currentUserId).eq("followingId", args.targetUserId)
          )
          .unique())
      : false

    // Check pending request
    const pendingRequest = currentUserId
      ? await ctx.db
          .query("followRequests")
          .withIndex("by_pair", (q) =>
            q.eq("fromUserId", currentUserId).eq("toUserId", args.targetUserId)
          )
          .unique()
      : null

    const hasRequestedFollow =
      pendingRequest?.status === "pending" ?? false

    // Check target account privacy
    const targetProfile = await ctx.db
      .query("userProfiles")
      .withIndex("by_user", (q) => q.eq("userId", args.targetUserId))
      .first()

    const isPrivate = targetProfile?.isPrivate ?? false

    const followersCount = await ctx.db
      .query("follows")
      .withIndex("by_following", (q) => q.eq("followingId", args.targetUserId))
      .collect()
      .then((rows) => rows.length)

    const followingCount = await ctx.db
      .query("follows")
      .withIndex("by_follower", (q) => q.eq("followerId", args.targetUserId))
      .collect()
      .then((rows) => rows.length)

    return {
      isFollowing,
      hasRequestedFollow,
      isPrivate,
      followersCount,
      followingCount,
    }
  },
})
```

#### getPendingFollowRequests query (for notifications):
```typescript
export const getPendingFollowRequests = query({
  args: {},
  handler: async (ctx) => {
    const currentUserId = await getCurrentUserId(ctx)
    if (!currentUserId) return []

    const requests = await ctx.db
      .query("followRequests")
      .withIndex("by_to", (q) => q.eq("toUserId", currentUserId))
      .filter((q) => q.eq(q.field("status"), "pending"))
      .collect()

    return Promise.all(
      requests.map(async (req) => {
        const userDoc = await ctx.db
          .query("users")
          .withIndex("by_userId", (q) => q.eq("userId", req.fromUserId))
          .first()

        return {
          _id: req._id,
          fromUserId: req.fromUserId,
          fromName: userDoc?.name ?? null,
          fromImage: userDoc?.image ?? null,
          createdAt: req.createdAt,
        }
      })
    )
  },
})
```

```typescript
const onToggleFollow = async () => {
  try {
    const result = await followUser({ targetUserId: userId })
    
    if (result.action === "followed") {
      setFollowing(true)
      setFollowers((prev) => prev + 1)
    } else if (result.action === "unfollowed") {
      setFollowing(false)
      setFollowers((prev) => prev - 1)
    } else if (result.action === "request_sent") {
      setRequested(true)
    } else if (result.action === "request_cancelled") {
      setRequested(false)
    }
  } catch (error) {

}
```
